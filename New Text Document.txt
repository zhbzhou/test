#!/bin/bash

# ============================================
# Tizen CTS Runner - Stable & Automated Version
# ============================================

SCRIPT_DIR=$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)
TIMESTAMP=$(date +%Y%m%d_%H%M%S)
RESULT_DIR="${SCRIPT_DIR}/results_${TIMESTAMP}"

mkdir -p "${RESULT_DIR}/qpa" "${RESULT_DIR}/csv" "${RESULT_DIR}/logs"

# 配置
WAYLAND_DISPLAY="wayland-0"
BATCH_SIZE=200
DEVICE_TEMP_DIR="/tmp/cts_temp"
DEVICE_QPA_DIR="/opt/media/USBDriveA1"
EXECUTOR_PATH="/usr/bin/VK-GL-CTS-1.4.5"

TEST_SUITES=(
    "egl-main-2020-03-01.txt:deqp-egl:egl"
    "gles2-main-2020-03-01.txt:deqp-gles2:gles2"
    "gles3-main-2020-03-01.txt:deqp-gles3:gles3"
    "gles31-main-2020-03-01.txt:deqp-gles31:gles31"
)

g_total_pass=0; g_total_fail=0; g_total_ns=0; g_total_all=0

analyze_qpa_local() {
    local qpa=$1
    [ ! -f "$qpa" ] && echo "0 0 0" && return
    awk -F'"' '/StatusCode=/ {
        s=$2;
        if(s=="Pass" || s=="QualityWarning") p++;
        else if(s=="Fail" || s=="ResourceError" || s=="InternalError" || s=="Crash") f++;
        else if(s=="NotSupported") n++;
    } END { print p+0, f+0, n+0 }' "$qpa"
}

# 环境清理
sdb shell "mkdir -p ${DEVICE_TEMP_DIR} ${DEVICE_QPA_DIR}"
sdb shell "rm -f ${DEVICE_QPA_DIR}/*.qpa ${DEVICE_QPA_DIR}/*.csv"

for suite_config in "${TEST_SUITES[@]}"; do
    IFS=':' read -r case_file binary_name suite_name <<< "$suite_config"
    CASE_PATH="${SCRIPT_DIR}/android/cts/main/${case_file}"
    [ ! -f "$CASE_PATH" ] && continue

    echo -e "\n\033[32m>>> Starting Suite: ${suite_name} <<<\033[0m"
    grep -vE '^(#|[[:space:]]*$)' "$CASE_PATH" > "${RESULT_DIR}/current_suite.txt"
    mapfile -t ALL_CASES < "${RESULT_DIR}/current_suite.txt"
    total_cases=${#ALL_CASES[@]}
    
    current_idx=0
    group_id=0

    while [ $current_idx -lt $total_cases ]; do
        remaining=$((total_cases - current_idx))
        actual_batch=$(( remaining < BATCH_SIZE ? remaining : BATCH_SIZE ))
        
        batch_file="${RESULT_DIR}/logs/batch_${suite_name}_g${group_id}.txt"
        printf "%s\n" "${ALL_CASES[@]:$current_idx:$actual_batch}" > "$batch_file"
        
        # 推送用例
        sdb push "$batch_file" "${DEVICE_TEMP_DIR}/run.txt" > /dev/null
        
        qpa_name="${suite_name}_g${group_id}.qpa"
        echo "Running Batch $group_id ($current_idx/$total_cases)..."
        
        # --- 关键修复：使用 -n 和重定向 < /dev/null 防止阻塞 ---
        # 同时将 cat 用例文件的工作放在设备端 shell 内部，确保 stdin 链条完整
        sdb shell -n "export XDG_RUNTIME_DIR=/run && export WAYLAND_DISPLAY=${WAYLAND_DISPLAY} && \
              cd ${EXECUTOR_PATH}/modules/${suite_name} && \
              cat ${DEVICE_TEMP_DIR}/run.txt | ./${binary_name} --deqp-stdin-caselist --deqp-log-filename=${DEVICE_QPA_DIR}/${qpa_name}" < /dev/null > /dev/null 2>&1
        
        # 拉取并分析以检测 Crash
        local_qpa="${RESULT_DIR}/qpa/${qpa_name}"
        sdb pull "${DEVICE_QPA_DIR}/${qpa_name}" "$local_qpa" > /dev/null 2>&1
        
        read p f n < <(analyze_qpa_local "$local_qpa")
        run_count=$((p + f + n))

        if [ $run_count -lt $actual_batch ]; then
            echo -e "\033[31m  Crash detected! Skipped index $((current_idx + run_count))\033[0m"
            current_idx=$((current_idx + run_count + 1))
        else
            current_idx=$((current_idx + actual_batch))
        fi
        group_id=$((group_id + 1))
        
        # 即使没有回车也给系统一点呼吸时间，防止 SDB 并发竞争
        sleep 1
    done

    # 设备端批量转换 CSV
    echo "Merging QPA results to CSV on device..."
    dev_csv="${DEVICE_QPA_DIR}/${suite_name}_final.csv"
    
    # 这里的 su -c 内部也要注意重定向
    convert_cmd="
        first=1
        echo 'CaseName,Result' > $dev_csv
        for qpa in ${DEVICE_QPA_DIR}/${suite_name}_g*.qpa; do
            if [ -f \"\$qpa\" ]; then
                if [ \$first -eq 1 ]; then
                    ${EXECUTOR_PATH}/executor/testlog-to-csv \$qpa >> $dev_csv
                    first=0
                else
                    ${EXECUTOR_PATH}/executor/testlog-to-csv \$qpa | tail -n +2 >> $dev_csv
                fi
            fi
        done
    "
    sdb shell -n "su -c '$convert_cmd'" < /dev/null

    local_csv="${RESULT_DIR}/csv/${suite_name}_final.csv"
    sdb pull "$dev_csv" "$local_csv" > /dev/null 2>&1

    # 统计并报告
    sp=$(awk -F',' '$2=="Pass"||$2=="QualityWarning"{c++}END{print c+0}' "$local_csv")
    sf=$(awk -F',' '$2=="Fail"||$2=="ResourceError"||$2=="InternalError"||$2=="Crash"{c++}END{print c+0}' "$local_csv")
    sn=$(awk -F',' '$2=="NotSupported"{c++}END{print c+0}' "$local_csv")
    
    echo "-------------------------------------------"
    echo "Suite ${suite_name} : Pass $sp, Fail $sf, Skip $sn"
    echo "-------------------------------------------"

    g_total_pass=$((g_total_pass + sp)); g_total_fail=$((g_total_fail + sf))
    g_total_ns=$((g_total_ns + sn)); g_total_all=$((g_total_all + total_cases))

    # 清理
    sdb shell -n "rm -f ${DEVICE_QPA_DIR}/${suite_name}_g*.qpa ${dev_csv}" < /dev/null
done

echo -e "\n==========================================="
echo "ALL DONE. Pass Rate: $(awk "BEGIN {printf \"%.2f\", ($g_total_pass*100)/($g_total_all+0.0001)}")%"
echo "==========================================="
