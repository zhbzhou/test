#!/bin/bash

# ============================================
# Optimized Tizen OpenGL/EGL CTS Runner (V2)
# Features: On-device Merge, Auto-Resume, High Efficiency
# ============================================

SCRIPT_DIR=$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)
TIMESTAMP=$(date +%Y%m%d_%H%M%S)
RESULT_DIR="${SCRIPT_DIR}/results_${TIMESTAMP}"

# 基础目录创建
mkdir -p "${RESULT_DIR}/qpa" "${RESULT_DIR}/csv" "${RESULT_DIR}/logs"

# 配置参数
WAYLAND_DISPLAY="wayland-0"
BATCH_SIZE=200            # 每个批次运行的用例数
DEVICE_TEMP_DIR="/tmp/cts_temp"
DEVICE_QPA_DIR="/opt/media/USBDriveA1"
EXECUTOR_PATH="/usr/bin/VK-GL-CTS-1.4.5"

TEST_SUITES=(
    "egl-main-2020-03-01.txt:deqp-egl:egl"
    "gles2-main-2020-03-01.txt:deqp-gles2:gles2"
    "gles3-main-2020-03-01.txt:deqp-gles3:gles3"
    "gles31-main-2020-03-01.txt:deqp-gles31:gles31"
)

# 初始化全局统计
g_total_pass=0; g_total_fail=0; g_total_ns=0; g_total_all=0

# ============================================
# 辅助工具函数
# ============================================

# 在 PC 端分析拉取回来的 QPA 状态
analyze_qpa_local() {
    local qpa=$1
    [ ! -f "$qpa" ] && echo "0 0 0" && return
    awk -F'"' '/StatusCode=/ {
        s=$2;
        if(s=="Pass" || s=="QualityWarning") p++;
        else if(s=="Fail" || s=="ResourceError" || s=="InternalError" || s=="Crash") f++;
        else if(s=="NotSupported") n++;
    } END { print p+0, f+0, n+0 }' "$qpa"
}

# 检查设备连接
check_conn() {
    sdb get-serialno >/dev/null 2>&1 || return 1
}

# ============================================
# 主执行逻辑
# ============================================

# 初始化设备环境
sdb shell "mkdir -p ${DEVICE_TEMP_DIR} ${DEVICE_QPA_DIR}"
sdb shell "rm -f ${DEVICE_QPA_DIR}/*.qpa ${DEVICE_QPA_DIR}/*.csv"

for suite_config in "${TEST_SUITES[@]}"; do
    IFS=':' read -r case_file binary_name suite_name <<< "$suite_config"
    CASE_PATH="${SCRIPT_DIR}/android/cts/main/${case_file}"
    
    [ ! -f "$CASE_PATH" ] && echo "Skip: $case_file not found" && continue

    echo -e "\n\033[32m>>> Starting Suite: ${suite_name} <<<\033[0m"
    
    # 提取有效用例
    grep -vE '^(#|[[:space:]]*$)' "$CASE_PATH" > "${RESULT_DIR}/current_suite.txt"
    mapfile -t ALL_CASES < "${RESULT_DIR}/current_suite.txt"
    total_cases=${#ALL_CASES[@]}
    
    current_idx=0
    group_id=0
    suite_start=$(date +%s)

    # --- 1. 测试执行循环 ---
    while [ $current_idx -lt $total_cases ]; do
        remaining=$((total_cases - current_idx))
        actual_batch=$(( remaining < BATCH_SIZE ? remaining : BATCH_SIZE ))
        
        batch_file="${RESULT_DIR}/logs/batch_${suite_name}_g${group_id}.txt"
        printf "%s\n" "${ALL_CASES[@]:$current_idx:$actual_batch}" > "$batch_file"
        
        sdb push "$batch_file" "${DEVICE_TEMP_DIR}/run.txt" > /dev/null
        
        qpa_name="${suite_name}_g${group_id}.qpa"
        echo "Running ${suite_name} | Batch $group_id | Index $current_idx/$total_cases..."
        
        # 核心命令：运行 CTS
        sdb shell "export XDG_RUNTIME_DIR=/run && export WAYLAND_DISPLAY=${WAYLAND_DISPLAY} && cd ${EXECUTOR_PATH}/modules/${suite_name} && ./${binary_name} --deqp-stdin-caselist --deqp-log-filename=${DEVICE_QPA_DIR}/${qpa_name}" > /dev/null 2>&1
        
        # 局部拉取分析（仅为了检测 Crash 和实时进度报表）
        local_qpa="${RESULT_DIR}/qpa/${qpa_name}"
        sdb pull "${DEVICE_QPA_DIR}/${qpa_name}" "$local_qpa" > /dev/null 2>&1
        
        read p f n < <(analyze_qpa_local "$local_qpa")
        run_count=$((p + f + n))

        if [ $run_count -lt $actual_batch ]; then
            # 发生 Crash：跳过崩溃点，从下一个开始
            echo -e "\033[31m  ⚠️  Crash detected at index $((current_idx + run_count)). Resuming...\033[0m"
            current_idx=$((current_idx + run_count + 1))
        else
            current_idx=$((current_idx + actual_batch))
        fi
        group_id=$((group_id + 1))
    done

    # --- 2. 设备端 CSV 转换与合并 ---
    echo "Processing CSV conversion on device..."
    dev_csv="${DEVICE_QPA_DIR}/${suite_name}_final.csv"
    
    # 构建设备端执行脚本：合并所有 QPA 并转为单个 CSV
    # 第一个文件保留表头，后续文件 tail -n +2 去除表头
    convert_cmd="
        first=1
        echo 'CaseName,Result' > $dev_csv
        for qpa in ${DEVICE_QPA_DIR}/${suite_name}_g*.qpa; do
            if [ -f \"\$qpa\" ]; then
                if [ \$first -eq 1 ]; then
                    ${EXECUTOR_PATH}/executor/testlog-to-csv \$qpa >> $dev_csv
                    first=0
                else
                    ${EXECUTOR_PATH}/executor/testlog-to-csv \$qpa | tail -n +2 >> $dev_csv
                fi
            fi
        done
    "
    sdb shell "su -c '$convert_cmd'"
    
    # 拉取最终 CSV
    local_csv="${RESULT_DIR}/csv/${suite_name}_final.csv"
    sdb pull "$dev_csv" "$local_csv" > /dev/null 2>&1

    # --- 3. 崩溃补偿：补齐缺失记录 ---
    csv_count=$(awk -F',' 'NR>1{c++}END{print c+0}' "$local_csv")
    if [ "$csv_count" -lt "$total_cases" ]; then
        echo "Patching $((total_cases - csv_count)) missing cases as Fail..."
        declare -A recorded
        while IFS=',' read -r name res; do recorded["$name"]=1; done < <(awk -F',' 'NR>1{print $1}' "$local_csv")
        for ((i=0; i<total_cases; i++)); do
            case_name="${ALL_CASES[$i]}"
            if [ -z "${recorded[$case_name]}" ]; then
                echo "${case_name},Fail" >> "$local_csv"
            fi
        done
        unset recorded
    fi

    # --- 4. 统计 Suite 报告 ---
    read sp sf sn < <(analyze_qpa_local "$local_csv") # 借用逻辑统计 CSV 里的状态
    # 重新从 CSV 统计更准确
    sp=$(awk -F',' '$2=="Pass"||$2=="QualityWarning"{c++}END{print c+0}' "$local_csv")
    sf=$(awk -F',' '$2=="Fail"||$2=="ResourceError"||$2=="InternalError"||$2=="Crash"{c++}END{print c+0}' "$local_csv")
    sn=$(awk -F',' '$2=="NotSupported"{c++}END{print c+0}' "$local_csv")
    
    suite_end=$(date +%s)
    echo "-------------------------------------------"
    echo "Suite ${suite_name} Done | Time: $((suite_end-suite_start))s"
    echo "Pass: $sp | Fail: $sf | NotSupported: $sn | Total: $total_cases"
    echo "-------------------------------------------"

    g_total_pass=$((g_total_pass + sp))
    g_total_fail=$((g_total_fail + sf))
    g_total_ns=$((g_total_ns + sn))
    g_total_all=$((g_total_all + total_cases))

    # 清理设备端该 Suite 的 QPA
    sdb shell "rm -f ${DEVICE_QPA_DIR}/${suite_name}_g*.qpa ${dev_csv}"
done

# ============================================
# 最终汇总报告
# ============================================
echo -e "\n\033[33m==========================================="
echo "         FINAL TEST SUMMARY"
echo "==========================================="
echo "TOTAL CASES:   $g_total_all"
echo "PASSED:        $g_total_pass"
echo "FAILED:        $g_total_fail"
echo "NOT SUPPORTED: $g_total_ns"
if [ $g_total_all -gt 0 ]; then
    rate=$(awk "BEGIN {printf \"%.2f\", ($g_total_pass*100)/$g_total_all}")
    echo "TOTAL PASS RATE: ${rate}%"
fi
echo -e "===========================================\033[0m\n"
